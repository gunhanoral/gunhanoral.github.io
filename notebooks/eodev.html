    ages = {
         1:  "Under 18",
        18:  "18-24",
        25:  "25-34",
        35:  "35-44",
        45:  "45-49",
        50:  "50-55",
        56:  "56+"
    }
    occupation = {
         0:  "other",
         1:  "academic/educator",
         2:  "artist",
         3:  "clerical/admin",
         4:  "college/grad student",
         5:  "customer service",
         6:  "doctor/health care",
         7:  "executive/managerial",
         8:  "farmer",
         9:  "homemaker",
        10:  "K-12 student",
        11:  "lawyer",
        12:  "programmer",
        13:  "retired",
        14:  "sales/marketing",
        15:  "scientist",
        16:  "self-employed",
        17:  "technician/engineer",
        18:  "tradesman/craftsman",
        19:  "unemployed",
        20:  "writer"
    }
    
    def readfiles(file, enc='utf-8'):
        with open(file, 'r', encoding=enc) as f:
            return f.read().splitlines()
    
    movies = readfiles('data/eodev/movies.dat', enc='iso-8859-15')
    ratings = readfiles('data/eodev/ratings.dat')
    users = readfiles('data/eodev/users.dat')
    
    moviesd = {}
    for line in movies:
        moviesd.update({line.split('::')[0]: {
            'title': line.split('::')[1], 
            'genres': line.split('::')[2]
        }})

    usersd = {}
    for line in users:
        usersd.update({line.split('::')[0]: {
            'gender': line.split('::')[1], 
            'age': ages[int(line.split('::')[2])],
            'occupation': occupation[int(line.split('::')[3])],
            'zip': line.split('::')[4]

        }})
    
    with open('data/eodev/merged.csv', 'w') as f:
        for line in ratings:
            user_id, movie_id, rating, timestamp = line.split('::')
            f.write("::".join([user_id, 
                  movie_id, 
                  rating, 
                  timestamp, 
                  usersd[user_id]['gender'], 
                  usersd[user_id]['age'], 
                  usersd[user_id]['occupation'], 
                  usersd[user_id]['zip'], 
                  moviesd[movie_id]['title'], 
                  moviesd[movie_id]['genres']]) + '\n'
            )

from collections import Counter
def problem2(file):

        years = (re.search('\((\d+)\)', line.split('::')[1]).group(1) for line in f.readlines())

def problem3(file):
    with open(file, 'r', encoding='iso-8859-15') as f:
        years = (genre for line in f.read().splitlines() for genre in line.split('::')[2].split('|'))
    print(' Genre Type', '{:>11}'.format('Occurence'))
    for genre, count in Counter(years).most_common():
        print('{:>11} {:>11}'.format(genre, count))

from operator import itemgetter
from itertools import groupby
def problem4(file):
    print('{:>11} {:>11}'.format('AgeRange', 'RatingC'))
    with open(file) as f:
        lines = f.read().splitlines()
        for k, v in groupby(sorted(((line.split('::')[5], line.split('::')[2]) for line in lines)), key=itemgetter(0)):
            print('{:>11} {:>11}'.format(k, len([i for x,i in v])))
